; Script templates fot VTS
; Each SECTION becomes a choice on the Insert menu.
; Each ITEM becomes a choice in the template selection dialog.
; The text lines between ITEM and ENDITEM are inserted into the script.
SECTION Service templates

ITEM ReadProperty
  SEND (
    DA = IUT_IP
    PDU = Confirmed-Request
    Service = ReadProperty
    Object = 0, object-type, instance
    Property = 1, property_id
    Unsigned = 2, index_if_you_need_it
  )

  EXPECT (
    SA  = IUT_IP   
    PDU = ComplexAck
    Service = ReadProperty
    Object = 0, OBJECT1
    Property = 1, Object-Identifier
    Unsigned = 2, index_if_you_need_it
    OpenTag 3
      -- your data here
    CloseTag 3
  )
ENDITEM

ITEM WriteProperty
  SEND (
    DA = IUT_IP
    PDU = Confirmed-Request
    Service = WriteProperty
    Object = 0, object-type, instance
    Property = 1, property_id
    Unsigned = 2, index_if_you_need_it
    OpenTag 3
      -- your data here
    CloseTag 3
    Unsigned = 4, priority_if_you_need_it
  )

  EXPECT (
    SA  = IUT_IP   
    PDU = SimpleAck
    Service = WriteProperty
  )
ENDITEM
